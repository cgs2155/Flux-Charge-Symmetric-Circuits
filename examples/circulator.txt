# Manuscript circulator: Josephson junction, two capacitors, one gyrator.
title Circulator

# two-terminal elements:  TYPE  name  node1 node2  value
J    e1  v1 v2  E_J
C    e2  v2 v3  C
C    e3  v3 v1  C

# gyrator:  gyrator  edge1 n1a n1b   edge2 n2a n2b   ratio
gyrator  e4 v1 v3   e5 v2 v3   G

# loops (faces of the planar circuit)
loop  f1  +e3 +e4
loop  f2  +e1 -e4 +e5
loop  f3  +e2 -e5
loop  f4  -e1 -e2 -e3

# gauge
ground v1
open   f4
